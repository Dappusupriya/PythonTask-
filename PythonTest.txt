class Test:
    def firstTest(self, filename):
        # Read the file content once
        with open(filename, 'r') as file:
            lines = file.readlines()
        
        # Task 4: Make a list of tuples (index, string)
        tuple_list = [(index, line.strip()) for index, line in enumerate(lines)]
        
        # Task 5: Make a list of dictionaries with keys and values
        dict_list = [
            {
                "index": index,
                "text": line,
                "total_words": len(line.split()),
                "total_characters": len(line.replace(" ", ""))  # Exclude spaces for character count
            }
            for index, line in tuple_list
        ]
        
        return tuple_list, dict_list
